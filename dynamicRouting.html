Question 1❓

Q: what is dynamic routing ?

✅ Answer

changing the data of url(like id/name etc) is called dynamic routing.
i have 100 employees and when i click one employe the id/name of employee should change in url and i go to anohter page and i can see all the details of this employe,It process is called dynamic routing.

the page on which we have to show the detail of user/employee in routes , the path of this page we set like path='/detail/:uid', see all process in example

💡 Example


all users page 

    //about.jsx

import React from "react";
import jsonUsers from "../utils/jsonUser";
import { useNavigate } from "react-router-dom";
import "./about.css";

const About = () => {

  const navigate = useNavigate();

  const viewUserHandler = (item) => {
    console.log(item);
    navigate(`/contact/${item.id}`);
  };

  return (
    <div className="about-container">

      <div className="about-wrapper">

        <h1 className="about-heading">
          Users List
        </h1>

        <div className="users-grid">

          {
            jsonUsers?.map((item, index) => {
              return (
                <div className="user-card" key={index}>

                  <div className="user-top">

                    <div className="user-image">
                      {item.name.charAt(0)}
                    </div>

                    <div className="user-info">
                      <h2>{item.name}</h2>
                      <p>@{item.username}</p>
                    </div>

                  </div>

                  <div className="user-details">
                    <span>{item.email}</span>
                  </div>

                  <button
                    className="view-btn"
                    onClick={() => viewUserHandler(item)}
                  >
                    View User
                  </button>

                </div>
              );
            })
          }

        </div>

      </div>

    </div>
  );
};

export default About;

this is user detail page where we go to see detail of specific user 

    //contact.jsx
import React from "react";
import Header from "../components/header/header";
import { useParams } from "react-router-dom";
import { useNavigate } from "react-router-dom";
import jsonUsers from "../utils/jsonUser";
import './contact.css';
const Contact = ()=>{
    const {uid} = useParams();
    // console.log(uid) 
    const navigate = useNavigate()
    const requiredUser = jsonUsers?.find((item,index)=> item.id == uid)
    return(
      <div className="main-container">

      <div className="profile-card">

        <div className="profile-top">

          <div className="profile-image">
            {requiredUser.name.charAt(0)}
          </div>

          <div className="profile-text">

            <p>@{requiredUser.username}</p>

            <span>Email</span>

          <div className="detail-box">
            <span>Street</span>
            <h3>{requiredUser.address.street}</h3>
          </div>

          <div className="detail-box">
            <span>Suite</span>
            <h3>{requiredUser.address.suite}</h3>
          </div>

          <div className="detail-box">
            <span>City</span>
            <h3>{requiredUser.address.city}</h3>
          </div>

          <div className="detail-box">
            <span>Zip Code</span>

        </div>

      </div>

    </div>
    )
}
export default Contact;
 
we define routes in dynamic routing like 

    //routes.jsx
import React from "react";
import {Routes,Route} from 'react-router-dom';
import Home from "../pages/home";
import About from "../pages/about";
import Contact from "../pages/contact";

const AppRoutes = ()=>{
    return(
        <Routes>
            <Route element={<Home/>} path='/' />
             <Route element={<About/>} path='/about' />

            <Route element={<Contact/>} path='/contact/:uid' />

        </Routes>
    )
}
export default  AppRoutes;
