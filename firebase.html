import React, { useState } from 'react'
import { app,auth } from '../../uitls/routes/firebase/firebase';
import { createUserWithEmailAndPassword } from 'firebase/auth';

const SignUpPage = () => {
    const [email,setEmail] = useState('');
    const [password,setPassword] = useState('');
    const [message,setMessage] = useState('');

    const signUpHandler = async ()=>{
        if(!email || !password){
            alert('please fill all fields');
            return;
        }
        try{
            const userCredential = await createUserWithEmailAndPassword(auth,email,password);
            console.log('user has stored');
            alert('signup successful!');
            setEmail('');
            setPassword('');
        }
        catch(error){
            console.log(error.message);
            setMessage(error.message);
        }
    }

  return (
    <div>
        <h1>signup page</h1>
        <input type="email" placeholder='enter your email'
        onChange={(e)=>setEmail(e.target.value)} value={email}
        autoComplete='new-email' /><br/>
        <input type="password" placeholder='enter your password' 
        onChange={(e)=>setPassword(e.target.value)} value={password}
        autoComplete='new-password'/>
        <button onClick={signUpHandler}>signup</button>
        {message && <p>{message}</p>}
    </div>
  )
}

export default SignUpPage

Question 2❓

Q: what is this
mport { initializeApp } from "firebase/app";
initializeApp what is ?

✅ Answer

This is a function of firebase that connects our project to firebase and we keep the configuration of our project that we receive from firebase in parameter of initializeApp like
const app = initializeApp(firebaseConfig);
it tells firebase that in which project we are working it just creates a connection of our project with firebase.

Question 4❓

Q: what is
import { createUserWithEmailAndPassword } from 'firebase/auth';

✅ Answer

this is a firebase funtion that is used for signup purpose, we send inside it (auth,email and password),if our auth,emial and password is correct it stores in firebase if any error it give error,like email already in use etc,

Note

we can not send object in it but we can send email,password.

Question 5❓

✅ Answer

💡 Example


       import { initializeApp } from "firebase/app";
import { getAuth } from "firebase/auth";

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
