            <button onClick={increment}>Increase</button>
            <button onClick={decrement}>decrease</button>
        </div>
    )
}

function Home(){
    const [count,setCount] = useState(0);
    useEffect(()=>{
        console.log('timer started');
        const timer = setInterval(() => {
            
            setCount(count + 1)
