Question 4❓

Q: Example of useState( )?

✅ Answer

see in example:

💡 Example


        
        

import React, { useEffect, useState } from "react";
const Home = ()=>{
    // first make state for counter 
    const [count,setCount] = useState(0);
    // make increment function to increase in count 
    const increment = ()=>{
        setCount(count + 1)
    }
    // make decrement function to decrease in count 
    const decrement = ()=>{
        setCount(count - 1)
    }
    // make use effect for show data on console
    useEffect(()=>{
        console.log('count: ', count)
    },[count])
    return(
        <div> 
            <h1>Couter App</h1>
            <h2>count : {count}</h2>
            <button onClick={increment}>Increase</button>
            <button onClick={decrement}>decrease</button>
        </div>
    )
}
export default Home;

Question 5❓

Q: cleanup in reactJs?

✅ Answer

it is required when we leave page our function must stop and when we come back funtion should start working .

💡 Example


       
        import React, { useEffect, useState } from "react";
function Home(){
    const [count,setCount] = useState(0);
    useEffect(()=>{
        console.log('timer started');
        const timer = setInterval(() => {
            
            setCount(count + 1)
        }, 1000);
        // cleanup runs when page mounts 
        return()=>{
            console.log('timer stopped');
            clearInterval(timer)
        }
    },[])
    return(
        <div>
            <h1>
                Timer
            </h1>
            <h2>{count}</h2>
        </div>
    )
}

Q: apicall and show data on dom by fetch?

✅ Answer

see in example:

💡 Example


       import React, { useEffect, useState } from "react";
const Home = ()=>{
    const [datat,setData] = useState([]);
    
    const apiCall = async ()=>{
        const apiUrl = 'https://jsonplaceholder.typicode.com/users';
        try{const get = await fetch(apiUrl);
            const jsonData = await get?.json();
            console.log(jsonData);
            setData(jsonData)
           
         }
        catch(error){console.log(error.name)}

    }
    // useeffect 
    useEffect(()=>{
        apiCall();
        
    },[])
    return(
        <div>
            <ul>
                {datat.map((item,index)=>{
                    return(
                        <li key={index}>{item.id} : {item.name}</li>
                    )
                })}
            </ul>
        </div>
    )
}
export default Home;

Question 8❓

Q: use of loading/loader in apicall?

✅ Answer

see in example:

💡 Example


       

  import React, { useEffect, useState } from "react";
import axios from "axios";
const Home = ()=>{
    // create use state for showing data in dom 
    const [list,setList] = useState([])
    // create a loading usestate
    const [loading,isLoading] = useState(false)
    const apiCall = async ()=>{
        isLoading(true)
        const apiUrl = 'https://jsonplaceholder.typicode.com/users';
        try{const data  = await axios.get(apiUrl);
           
            console.log(data.data);
           
          
            if(data.data){
                setList(data.data);
                isLoading(false);
            }
        }
        catch(error){console.log(error)}
    }

    // useEffect 
    useEffect(()=>{
        apiCall();
    },[])
    return(
        <div>
        <h1>User List by axios</h1>
      {  loading && <h3>data is loading</h3>}
        <ul>
            {list.map((item,index)=>{
                return(
                    <li key={index}>{item.name}</li>
                )
            })}
        </ul>
        </div>
    )
}
export default Home;
