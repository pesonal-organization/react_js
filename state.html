Question 3â“

Q: Example of object useState?

âœ… Answer

we can store and update non-primitive data like array and object in useState also,

ğŸ’¡ Example


       
        import { useState } from "react";

const UseStateHook = () => {

  const [user, setUser] = useState({
    name: "Aziz",
    age: 30,
    city: "Karachi"
  });

  const increaseAge = () => {
    setUser({
      ...user,
      age: user.age + 1
    });
  };

  return (
    <>
      <h1>Example of Object State</h1>
      <h2>Name: {user.name}</h2>
      <h2>Age: {user.age}</h2>
      <h2>City: {user.city}</h2>

      <button onClick={increaseAge}>
        Increase Age
      </button>
    </>
  );
};

export default UseStateHook;
