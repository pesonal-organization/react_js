        practice.jsx

        
        // practice 
import React from "react";
import { useState,useEffect } from "react";
const Practice = ()=>{
    const [counter,setCounter] = useState(0);
    const [userName, setUserName] = useState('')
    // state for watch
const [date,setDate] = useState('');
let timeHandler;


    // updating counter 
    const update = ()=>{
        console.log('counter is updating')
    }
    // user name is update
    const updating = ()=>{console.log('username is updating')}
    
    
    // useEffect for counter updating
    useEffect(()=>{update()},[counter])
    // useEffect for userName 
    useEffect(()=>{updating()},[userName])
    // watch
    useEffect(()=>{
        timeHandler = setInterval(() => {
            const current = new Date().toLocaleTimeString();
            setDate(current)

            
        }, 1000);
        // when un mount 
        console.log('ok i stopped time');
        return()=>{clearInterval(timeHandler)}
    },[

    ])
    
    return(
    <div>

    <h1>I am practice screen</h1>
    <h1>counter : {counter ? counter :'No counter'}</h1>
    <h1>userName : {userName?userName:'No user name available'}</h1>
    <h1>{date}</h1>
    <button onClick={()=>setCounter(counter + 1)}>Increment</button>
    <button onClick={()=>setUserName('saqi')}>user Name</button>

    </div>
)}
export default Practice;
