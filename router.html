       router.jsx 
       
        // Note : All routes are defined here
import { Routes, Route} from "react-router-dom";
// import {BrowserRouter} from "react-router";
import Home from "../screens/home/home";
import About from "../screens/about/about";
import Contact from "../screens/contact/contact";

const RoutesComponent = ()=>{
    return(
    
        <Routes>
            <Route path="/" Component={Home}/>
            <Route path="/about" Component={About } />
            <Route path="/contact" Component={Contact} />
        </Routes>
       
    )
}
export default RoutesComponent;

Question 3❓

Q: what is screen in react-js?

✅ Answer

screen is the name of a folder that is kept inside src in reactjs folder, and inside screen folder we make multiple components like home, about, contact etc

Question 4❓

Q: what is utiles?

✅ Answer

utils is just a folder that is created inside src, and in utiles we keep the file that we have to use multiple places in project, normally we keep router.jsx file in utiles.
