// login screen
import React, { useState } from "react";
import Header from "../../components/header/header";
import { v4 as uuidv4 } from 'uuid';
import { useNavigate } from "react-router";
const Login = ()=>{
    const [email,setEmail] = useState();
    const [password,setPassword] = useState();
    const navigate = useNavigate();

    const submitForm = ()=>{
        const user = {email,
            password,
            uid:uuidv4()
        }
        console.log('user: ',user)
        localStorage.setItem('AuthState',JSON.stringify(user))
        alert('You have logged in successfully');
        window.location.reload();
    };
    return(
    <>
    <label htmlFor="">Email</label>
    <input type="email" name="" placeholder="enter your email" id=""value={email}
    onChange={(e)=>setEmail(e.target.value)}/>
    <label htmlFor="">password</label>
    <input type="password" value={password} 
    onChange={(e)=>setPassword(e.target.value)} />
    <button  onClick={submitForm}
    >sign in</button>
    </>
    )
}
export default Login;

routes.jsx 

// all routes are defined here



import { useEffect, useState } from 'react';
import PublicRoutes from './unprotected-routes';
import ProtectedRotes from './protected-routes';
import { useLocation ,useNavigate} from 'react-router';
import { publicRoutes,protectedRoutes } from '../utils/routes';

const auth = null;
const AppRoutes = ()=>{
    const [authStatus,setAuthStatus] = useState(null)
    const navigate = useNavigate();
    const {pathname} = useLocation();
    // authStatus checker hook
    useEffect(()=>{
        console.log('authstaus: ',authStatus);
        console.log('path: ',pathname)
        if(authStatus == null && protectedRoutes.includes(pathname)){
            navigate(publicRoutes[1])
        }
        // note getting AuthStats from local stroage
        const auth = localStorage.getItem('AuthState')
        const jsonAuth = JSON.parse(auth);
        console.log(jsonAuth);
        if(!jsonAuth){if(!authStatus && protectedRoutes.includes(pathname)) navigate(publicRoutes[1])}
        else if(jsonAuth){setAuthStatus(jsonAuth)
            navigate('/')
        }
    },[])
    return( 
        <>
       { (authStatus)
        ?
        (<ProtectedRotes/>)
        :
        (<PublicRoutes/>)}
        </>
    )
}
export default AppRoutes;
